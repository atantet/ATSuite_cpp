400  gsl_vector *bounds;
401  gsl_matrix_uint *gridMem;
402 
403  // Get grid size

405  bounds = gridBounds->at(dir);
406  N *= bounds->size - 1;
407  }
408  gridMem = gsl_matrix_uint_alloc(N, 2);
409 
410  // Assign a pair of source and destination boxes to each trajectory
411 #pragma omp parallel
412  {
413  gsl_vector *X = gsl_vector_alloc(dim);
414

416  for (size_t traj = 0; traj < nTraj; traj++) {
417  // Find initial box

419  gsl_matrix_uint_set(gridMem, traj, 0, getBoxMembership(X, gridBounds));

423  gsl_matrix_uint_set(gridMem, traj, 1, getBoxMembership(X, gridBounds));
424  }
425  gsl_vector_free(X);
426  }

519 getBoxMembership(const gsl_vector *state, const std::vector<gsl_vector *> *gridBounds)
520 {
521  const size_t dim = state->size;
522  size_t inBox, nBoxDir;
523  size_t foundBox;
524  size_t subbp, subbn, ids;
525  gsl_vector *bounds;
526  size_t N = 1;
527 
528  // Get grid dimensions
529  for (size_t d = 0; d < dim; d++){
530  bounds = gridBounds->at(d);
531  N *= bounds->size - 1;
532  }
533 
534  // Get box
535  foundBox = N;
536  for (size_t box = 0; box < N; box++){
537  inBox = 0;
538  subbp = box;
539  for (size_t d = 0; d < dim; d++){
540  bounds = gridBounds->at(d);
541  nBoxDir = bounds->size - 1;
542  subbn = (size_t) (subbp / nBoxDir);
543  ids = subbp - subbn * nBoxDir;
544  inBox += (size_t) ((gsl_vector_get(state, d)
545  >= gsl_vector_get(bounds, ids))
546  & (gsl_vector_get(state, d)
547  < gsl_vector_get(bounds, ids+1)));
548  subbp = subbn;
549  }
550  if (inBox == dim){
551  foundBox = box;
552  break;
553  }
554  }
555 
556  return foundBox;
557 }
